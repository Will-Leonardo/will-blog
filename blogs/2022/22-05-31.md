<Boxx type='tip' />


```Vue
<template>
    <!-- 注意不需要用 :ref -->
    <el-input ref="dom" />
<template>
<script lang="ts">
import { defineComponent, onMounted, ref } from "vue";
export default defineComponent({
    setup() { 
        const dom = ref<HTMLElement|null>(null);
        onMounted(()=>{
           dom.value.focus();
        })
        return {
          dom
        }
    }
})
</script>
```
